import React, { useState, useEffect } from 'react';
import { Link } from 'react-router-dom';
import { createPageUrl } from '@/utils';
import { base44 } from '@/api/base44Client';
import { useQuery } from '@tanstack/react-query';
import { Button } from "@/components/ui/button";
import { Avatar, AvatarFallback, AvatarImage } from "@/components/ui/avatar";
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs";
import { Skeleton } from "@/components/ui/skeleton";
import GadgetCard from '@/components/gadgets/GadgetCard';
import SellerRating from '@/components/profile/SellerRating';
import { 
  MapPin, Calendar, Mail, Phone, Star, 
  Package, MessageCircle, Settings, ArrowLeft 
} from 'lucide-react';
import { format } from 'date-fns';
import { motion } from 'framer-motion';

export default function Profile() {
  const urlParams = new URLSearchParams(window.location.search);
  const profileEmail = urlParams.get('email');
  
  const [currentUser, setCurrentUser] = useState(null);
  const [isOwnProfile, setIsOwnProfile] = useState(false);

  useEffect(() => {
    const checkAuth = async () => {
      const isAuth = await base44.auth.isAuthenticated();
      if (isAuth) {
        const user = await base44.auth.me();
        setCurrentUser(user);
        setIsOwnProfile(!profileEmail || profileEmail === user.email);
      }
    };
    checkAuth();
  }, [profileEmail]);

  const targetEmail = profileEmail || currentUser?.email;

  const { data: profileUser, isLoading: loadingUser } = useQuery({
    queryKey: ['profile-user', targetEmail],
    queryFn: async () => {
      if (!targetEmail) return null;
      const users = await base44.entities.User.filter({ email: targetEmail });
      return users[0] || null;
    },
    enabled: !!targetEmail
  });

  const { data: listings = [], isLoading: loadingListings } = useQuery({
    queryKey: ['user-listings', targetEmail],
    queryFn: () => base44.entities.Gadget.filter({ seller_email: targetEmail }, '-created_date'),
    enabled: !!targetEmail
  });

  const { data: reviews = [] } = useQuery({
    queryKey: ['user-reviews', targetEmail],
    queryFn: () => base44.entities.Review.filter({ seller_email: targetEmail }),
    enabled: !!targetEmail
  });

  const averageRating = reviews.length > 0 
    ? reviews.reduce((sum, r) => sum + r.rating, 0) / reviews.length 
    : 0;

  const activeListings = listings.filter(l => l.status === 'available');
  const soldListings = listings.filter(l => l.status === 'sold');

  if (loadingUser && !currentUser) {
    return (
      <div className="min-h-screen bg-slate-50 py-8">
        <div className="max-w-4xl mx-auto px-4">
          <Skeleton className="h-48 rounded-2xl mb-6" />
          <div className="space-y-4">
            <Skeleton className="h-8 w-48" />
            <Skeleton className="h-4 w-64" />
          </div>
        </div>
      </div>
    );
  }

  const displayUser = profileUser || currentUser;

  if (!displayUser && !loadingUser) {
    return (
      <div className="min-h-screen bg-slate-50 flex items-center justify-center">
        <div className="text-center">
          <h2 className="text-xl font-semibold text-slate-900 mb-2">User not found</h2>
          <Link to={createPageUrl('Home')}>
            <Button>Back to Home</Button>
          </Link>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-slate-50">
      {/* Header */}
      <div className="bg-gradient-to-br from-violet-600 to-purple-700 pt-8 pb-24">
        <div className="max-w-4xl mx-auto px-4">
          <Link to={createPageUrl('Home')} className="inline-flex items-center text-sm text-white/70 hover:text-white mb-6">
            <ArrowLeft className="w-4 h-4 mr-1" />
            Back to listings
          </Link>
        </div>
      </div>

      {/* Profile Card */}
      <div className="max-w-4xl mx-auto px-4 -mt-20">
        <motion.div
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          className="bg-white rounded-2xl shadow-lg p-6 sm:p-8"
        >
          <div className="flex flex-col sm:flex-row items-start gap-6">
            <Avatar className="w-24 h-24 border-4 border-white shadow-lg">
              <AvatarImage src={displayUser?.avatar_url} />
              <AvatarFallback className="bg-violet-100 text-violet-600 text-2xl">
                {displayUser?.full_name?.[0]?.toUpperCase() || 'U'}
              </AvatarFallback>
            </Avatar>

            <div className="flex-1">
              <h1 className="text-2xl font-bold text-slate-900">
                {displayUser?.full_name || 'Anonymous User'}
              </h1>
              
              <SellerRating rating={averageRating} reviewCount={reviews.length} />
              
              <div className="flex flex-wrap gap-4 mt-4 text-sm text-slate-500">
                {displayUser?.location && (
                  <span className="flex items-center gap-1">
                    <MapPin className="w-4 h-4" />
                    {displayUser.location}
                  </span>
                )}
                <span className="flex items-center gap-1">
                  <Calendar className="w-4 h-4" />
                  Member since {displayUser?.created_date ? format(new Date(displayUser.created_date), 'MMM yyyy') : 'Recently'}
                </span>
              </div>

              {displayUser?.bio && (
                <p className="mt-4 text-slate-600">{displayUser.bio}</p>
              )}
            </div>

            <div className="flex gap-2">
              {isOwnProfile ? (
                <Link to={createPageUrl('Settings')}>
                  <Button variant="outline" className="gap-2">
                    <Settings className="w-4 h-4" />
                    Edit Profile
                  </Button>
                </Link>
              ) : (
                <a href={`mailto:${targetEmail}`}>
                  <Button className="bg-violet-600 hover:bg-violet-700 gap-2">
                    <MessageCircle className="w-4 h-4" />
                    Contact
                  </Button>
                </a>
              )}
            </div>
          </div>

          {/* Stats */}
          <div className="grid grid-cols-3 gap-4 mt-8 pt-8 border-t">
            <div className="text-center">
              <p className="text-2xl font-bold text-slate-900">{activeListings.length}</p>
              <p className="text-sm text-slate-500">Active Listings</p>
            </div>
            <div className="text-center">
              <p className="text-2xl font-bold text-slate-900">{soldListings.length}</p>
              <p className="text-sm text-slate-500">Items Sold</p>
            </div>
            <div className="text-center">
              <p className="text-2xl font-bold text-slate-900">{reviews.length}</p>
              <p className="text-sm text-slate-500">Reviews</p>
            </div>
          </div>
        </motion.div>

        {/* Tabs */}
        <Tabs defaultValue="listings" className="mt-8">
          <TabsList className="bg-white rounded-xl p-1 shadow-sm">
            <TabsTrigger value="listings" className="rounded-lg px-6">
              <Package className="w-4 h-4 mr-2" />
              Listings ({listings.length})
            </TabsTrigger>
            <TabsTrigger value="reviews" className="rounded-lg px-6">
              <Star className="w-4 h-4 mr-2" />
              Reviews ({reviews.length})
            </TabsTrigger>
          </TabsList>

          <TabsContent value="listings" className="mt-6">
            {loadingListings ? (
              <div className="grid grid-cols-2 sm:grid-cols-3 gap-4">
                {Array.from({ length: 6 }).map((_, i) => (
                  <Skeleton key={i} className="aspect-square rounded-2xl" />
                ))}
              </div>
            ) : listings.length === 0 ? (
              <div className="text-center py-12 bg-white rounded-2xl">
                <Package className="w-12 h-12 text-slate-300 mx-auto mb-4" />
                <p className="text-slate-500">No listings yet</p>
                {isOwnProfile && (
                  <Link to={createPageUrl('CreateListing')}>
                    <Button className="mt-4 bg-violet-600 hover:bg-violet-700">
                      Create Your First Listing
                    </Button>
                  </Link>
                )}
              </div>
            ) : (
              <div className="grid grid-cols-2 sm:grid-cols-3 gap-4">
                {listings.map((gadget, index) => (
                  <GadgetCard key={gadget.id} gadget={gadget} index={index} />
                ))}
              </div>
            )}
          </TabsContent>

          <TabsContent value="reviews" className="mt-6">
            {reviews.length === 0 ? (
              <div className="text-center py-12 bg-white rounded-2xl">
                <Star className="w-12 h-12 text-slate-300 mx-auto mb-4" />
                <p className="text-slate-500">No reviews yet</p>
              </div>
            ) : (
              <div className="space-y-4">
                {reviews.map((review) => (
                  <div key={review.id} className="bg-white rounded-xl p-5">
                    <div className="flex items-start gap-4">
                      <Avatar>
                        <AvatarFallback className="bg-slate-100">
                          {review.reviewer_name?.[0]?.toUpperCase() || 'R'}
                        </AvatarFallback>
                      </Avatar>
                      <div className="flex-1">
                        <div className="flex items-center justify-between">
                          <p className="font-semibold text-slate-900">{review.reviewer_name || 'Anonymous'}</p>
                          <div className="flex">
                            {Array.from({ length: 5 }).map((_, i) => (
                              <Star
                                key={i}
                                className={`w-4 h-4 ${i < review.rating ? 'text-amber-400 fill-amber-400' : 'text-slate-200'}`}
                              />
                            ))}
                          </div>
                        </div>
                        {review.comment && (
                          <p className="text-slate-600 mt-2">{review.comment}</p>
                        )}
                        <p className="text-xs text-slate-400 mt-2">
                          {review.created_date ? format(new Date(review.created_date), 'MMM d, yyyy') : ''}
                        </p>
                      </div>
                    </div>
                  </div>
                ))}
              </div>
            )}
          </TabsContent>
        </Tabs>
      </div>
    </div>
  );
}
