import React, { useState, useEffect } from 'react';
import { Link } from 'react-router-dom';
import { createPageUrl } from '@/utils';
import { base44 } from '@/api/base44Client';
import { useQuery } from '@tanstack/react-query';
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { Skeleton } from "@/components/ui/skeleton";
import { Avatar, AvatarFallback, AvatarImage } from "@/components/ui/avatar";
import SellerRating from '@/components/profile/SellerRating';
import PaymentModal from '@/components/payment/PaymentModal';
import { 
  ChevronLeft, ChevronRight, MapPin, Calendar, Eye, 
  MessageCircle, ShoppingCart, Share2, Heart, Shield,
  Phone, Mail, User
} from 'lucide-react';
import { motion, AnimatePresence } from 'framer-motion';
import { format } from 'date-fns';

const conditionLabels = {
  new: { label: 'Brand New', class: 'bg-emerald-500 text-white' },
  like_new: { label: 'Like New', class: 'bg-green-500 text-white' },
  good: { label: 'Good Condition', class: 'bg-blue-500 text-white' },
  fair: { label: 'Fair Condition', class: 'bg-yellow-500 text-white' },
  poor: { label: 'Poor Condition', class: 'bg-orange-500 text-white' }
};

export default function GadgetDetail() {
  const urlParams = new URLSearchParams(window.location.search);
  const gadgetId = urlParams.get('id');
  
  const [currentImage, setCurrentImage] = useState(0);
  const [showPayment, setShowPayment] = useState(false);
  const [currentUser, setCurrentUser] = useState(null);
  const [isLoggedIn, setIsLoggedIn] = useState(false);

  useEffect(() => {
    const checkAuth = async () => {
      const authenticated = await base44.auth.isAuthenticated();
      setIsLoggedIn(authenticated);
      if (authenticated) {
        const user = await base44.auth.me();
        setCurrentUser(user);
      }
    };
    checkAuth();
  }, []);

  const { data: gadget, isLoading } = useQuery({
    queryKey: ['gadget', gadgetId],
    queryFn: async () => {
      const gadgets = await base44.entities.Gadget.filter({ id: gadgetId });
      if (gadgets.length > 0) {
        // Increment views
        await base44.entities.Gadget.update(gadgetId, { 
          views: (gadgets[0].views || 0) + 1 
        });
        return gadgets[0];
      }
      return null;
    },
    enabled: !!gadgetId
  });

  const { data: sellerReviews = [] } = useQuery({
    queryKey: ['seller-reviews', gadget?.seller_email],
    queryFn: () => base44.entities.Review.filter({ seller_email: gadget.seller_email }),
    enabled: !!gadget?.seller_email
  });

  const sellerRating = sellerReviews.length > 0 
    ? sellerReviews.reduce((sum, r) => sum + r.rating, 0) / sellerReviews.length 
    : 0;

  const condition = conditionLabels[gadget?.condition] || conditionLabels.good;

  const nextImage = () => {
    if (gadget?.images?.length > 1) {
      setCurrentImage((prev) => (prev + 1) % gadget.images.length);
    }
  };

  const prevImage = () => {
    if (gadget?.images?.length > 1) {
      setCurrentImage((prev) => (prev - 1 + gadget.images.length) % gadget.images.length);
    }
  };

  const handleBuyNow = () => {
    if (!isLoggedIn) {
      base44.auth.redirectToLogin(window.location.href);
      return;
    }
    setShowPayment(true);
  };

  if (isLoading) {
    return (
      <div className="min-h-screen bg-slate-50 py-8">
        <div className="max-w-6xl mx-auto px-4">
          <div className="grid lg:grid-cols-2 gap-8">
            <Skeleton className="aspect-square rounded-2xl" />
            <div className="space-y-4">
              <Skeleton className="h-8 w-3/4" />
              <Skeleton className="h-12 w-1/3" />
              <Skeleton className="h-24" />
            </div>
          </div>
        </div>
      </div>
    );
  }

  if (!gadget) {
    return (
      <div className="min-h-screen bg-slate-50 flex items-center justify-center">
        <div className="text-center">
          <h2 className="text-xl font-semibold text-slate-900 mb-2">Gadget not found</h2>
          <Link to={createPageUrl('Home')}>
            <Button>Back to Home</Button>
          </Link>
        </div>
      </div>
    );
  }

  const isOwner = currentUser?.email === gadget.seller_email;

  return (
    <div className="min-h-screen bg-slate-50">
      {/* Breadcrumb */}
      <div className="bg-white border-b">
        <div className="max-w-6xl mx-auto px-4 py-3">
          <Link to={createPageUrl('Home')} className="inline-flex items-center text-sm text-slate-500 hover:text-violet-600">
            <ChevronLeft className="w-4 h-4 mr-1" />
            Back to listings
          </Link>
        </div>
      </div>

      <div className="max-w-6xl mx-auto px-4 py-8">
        <div className="grid lg:grid-cols-2 gap-8">
          {/* Image Gallery */}
          <div className="space-y-4">
            <div className="relative aspect-square bg-white rounded-2xl overflow-hidden shadow-sm">
              {gadget.images?.length > 0 ? (
                <>
                  <AnimatePresence mode="wait">
                    <motion.img
                      key={currentImage}
                      initial={{ opacity: 0 }}
                      animate={{ opacity: 1 }}
                      exit={{ opacity: 0 }}
                      src={gadget.images[currentImage]}
                      alt={gadget.title}
                      className="w-full h-full object-contain"
                    />
                  </AnimatePresence>
                  
                  {gadget.images.length > 1 && (
                    <>
                      <button
                        onClick={prevImage}
                        className="absolute left-4 top-1/2 -translate-y-1/2 w-10 h-10 bg-white/90 rounded-full flex items-center justify-center shadow-lg hover:bg-white"
                      >
                        <ChevronLeft className="w-5 h-5" />
                      </button>
                      <button
                        onClick={nextImage}
                        className="absolute right-4 top-1/2 -translate-y-1/2 w-10 h-10 bg-white/90 rounded-full flex items-center justify-center shadow-lg hover:bg-white"
                      >
                        <ChevronRight className="w-5 h-5" />
                      </button>
                    </>
                  )}
                </>
              ) : (
                <div className="w-full h-full flex items-center justify-center bg-slate-100">
                  <span className="text-slate-400">No image available</span>
                </div>
              )}
              
              {gadget.status === 'sold' && (
                <div className="absolute inset-0 bg-black/60 flex items-center justify-center">
                  <span className="text-white font-bold text-3xl tracking-wider">SOLD</span>
                </div>
              )}
            </div>
            
            {gadget.images?.length > 1 && (
              <div className="flex gap-2 overflow-x-auto pb-2">
                {gadget.images.map((img, i) => (
                  <button
                    key={i}
                    onClick={() => setCurrentImage(i)}
                    className={`flex-shrink-0 w-20 h-20 rounded-lg overflow-hidden border-2 transition-all ${
                      i === currentImage ? 'border-violet-500' : 'border-transparent'
                    }`}
                  >
                    <img src={img} alt="" className="w-full h-full object-cover" />
                  </button>
                ))}
              </div>
            )}
          </div>

          {/* Details */}
          <div className="space-y-6">
            <div>
              <div className="flex flex-wrap gap-2 mb-3">
                <Badge variant="outline" className="capitalize">{gadget.category}</Badge>
                <Badge className={condition.class}>{condition.label}</Badge>
                {gadget.status === 'reserved' && (
                  <Badge className="bg-amber-500">Reserved</Badge>
                )}
              </div>
              
              <h1 className="text-2xl sm:text-3xl font-bold text-slate-900 mb-2">
                {gadget.title}
              </h1>
              
              <div className="flex items-center gap-4 text-sm text-slate-500">
                <span className="flex items-center gap-1">
                  <Eye className="w-4 h-4" />
                  {gadget.views || 0} views
                </span>
                <span className="flex items-center gap-1">
                  <Calendar className="w-4 h-4" />
                  {gadget.created_date ? format(new Date(gadget.created_date), 'MMM d, yyyy') : 'Recently'}
                </span>
              </div>
            </div>

            <div className="text-4xl font-bold text-slate-900">
              ${gadget.price?.toLocaleString()}
            </div>

            {gadget.description && (
              <div>
                <h3 className="font-semibold text-slate-900 mb-2">Description</h3>
                <p className="text-slate-600 whitespace-pre-wrap">{gadget.description}</p>
              </div>
            )}

            {gadget.location && (
              <div className="flex items-center gap-2 text-slate-600">
                <MapPin className="w-4 h-4" />
                <span>{gadget.location}</span>
              </div>
            )}

            {/* Seller Info */}
            <div className="bg-white rounded-xl p-5 border border-slate-100">
              <h3 className="font-semibold text-slate-900 mb-4">Seller Information</h3>
              <div className="flex items-start gap-4">
                <Link to={createPageUrl(`Profile?email=${gadget.seller_email}`)}>
                  <Avatar className="w-14 h-14">
                    <AvatarFallback className="bg-violet-100 text-violet-600 text-lg">
                      {gadget.seller_name?.[0]?.toUpperCase() || 'S'}
                    </AvatarFallback>
                  </Avatar>
                </Link>
                <div className="flex-1">
                  <Link 
                    to={createPageUrl(`Profile?email=${gadget.seller_email}`)}
                    className="font-semibold text-slate-900 hover:text-violet-600"
                  >
                    {gadget.seller_name || 'Anonymous'}
                  </Link>
                  <SellerRating rating={sellerRating} reviewCount={sellerReviews.length} size="small" />
                  
                  {gadget.seller_phone && (
                    <div className="flex items-center gap-2 text-sm text-slate-500 mt-2">
                      <Phone className="w-3.5 h-3.5" />
                      <span>{gadget.seller_phone}</span>
                    </div>
                  )}
                </div>
              </div>
            </div>

            {/* Action Buttons */}
            {gadget.status === 'available' && !isOwner && (
              <div className="flex gap-3">
                <Button 
                  size="lg" 
                  className="flex-1 bg-violet-600 hover:bg-violet-700 h-14 rounded-xl font-semibold"
                  onClick={handleBuyNow}
                >
                  <ShoppingCart className="w-5 h-5 mr-2" />
                  Buy Now
                </Button>
                <a href={`mailto:${gadget.seller_email}?subject=Inquiry about ${gadget.title}`}>
                  <Button size="lg" variant="outline" className="h-14 rounded-xl px-6">
                    <MessageCircle className="w-5 h-5" />
                  </Button>
                </a>
              </div>
            )}

            {isOwner && (
              <div className="bg-violet-50 border border-violet-100 rounded-xl p-4">
                <p className="text-sm text-violet-700 mb-3">This is your listing</p>
                <Link to={createPageUrl(`EditListing?id=${gadget.id}`)}>
                  <Button variant="outline" className="w-full">Edit Listing</Button>
                </Link>
              </div>
            )}

            {/* Trust Badges */}
            <div className="flex items-center gap-3 text-sm text-slate-500">
              <Shield className="w-5 h-5 text-green-500" />
              <span>Secure payments with PayPal & EcoCash</span>
            </div>
          </div>
        </div>
      </div>

      {/* Payment Modal */}
      {showPayment && currentUser && (
        <PaymentModal
          open={showPayment}
          onClose={() => setShowPayment(false)}
          gadget={gadget}
          seller={{ email: gadget.seller_email, name: gadget.seller_name }}
          buyer={currentUser}
          onSuccess={() => window.location.reload()}
        />
      )}
    </div>
  );
}
